#include <stdio.h>
#include <string.h>

int main()
{
    char prod1[] = "iEtS";
    char prod2[] = "iEtSeS";
    char prefix[20];
    int i = 0;

    printf("Given Grammar:\n");
    printf("S -> iEtS | iEtSeS | a\n");
    printf("E -> b\n");

    // Find common prefix
    while(prod1[i] == prod2[i] && prod1[i] != '\0')
    {
        prefix[i] = prod1[i];
        i++;
    }
    prefix[i] = '\0';

    printf("\nGrammar After Left Factoring:\n");
    printf("S  -> %sS' | a\n", prefix);

    printf("S' -> %s | Îµ\n", prod2 + i);

    printf("E  -> b\n");

    return 0;
}
