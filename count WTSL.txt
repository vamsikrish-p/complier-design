#include <stdio.h>

int main() {
    int ch;
    int words = 0, lines = 0, spaces = 0, tabs = 0;
    int inWord = 0;

    printf("Enter text (Press Ctrl+D to end):\n");

    while ((ch = getchar()) != EOF) {

        if (ch == '\n')
            lines++;

        if (ch == ' ')
            spaces++;

        if (ch == '\t')
            tabs++;

        if (ch == ' ' || ch == '\n' || ch == '\t')
            inWord = 0;
        else if (inWord == 0) {
            inWord = 1;
            words++;
        }
    }

    printf("\nLines: %d", lines);
    printf("\nWords: %d", words);
    printf("\nSpaces: %d", spaces);
    printf("\nTabs: %d\n", tabs);

    return 0;
}
